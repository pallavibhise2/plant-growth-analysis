// 5 Main MeasuresAvg_Humidity = AVERAGE('plant_growth_data'[Humidity])Avg_Sunlight_Hours = AVERAGE('plant_growth_data'[Sunlight_Hours])Avg_Temperature = AVERAGE('plant_growth_data'[Temperature])Growth_Milestone_Count = COUNTROWS('plant_growth_data')Growth_Percentage = DIVIDE(    [Growth_Milestone_Count],    CALCULATE([Growth_Milestone_Count], ALL('plant_growth_data')))// 7 Calculated ColumnsWater_Frequency_Numeric = SWITCH(    'plant_growth_data'[Water_Frequency],    "daily", 1,    "weekly", 2,    "bi-weekly", 3,    0)Temperature_Range = SWITCH(    TRUE(),    'plant_growth_data'[Temperature] < 20, "Cold",    'plant_growth_data'[Temperature] <= 30, "Moderate",    "Warm")Humidity_Range = SWITCH(    TRUE(),    'plant_growth_data'[Humidity] < 40, "Low",    'plant_growth_data'[Humidity] <= 70, "Moderate",    "High")Humidity_Level_Description = SWITCH(    TRUE(),    'plant_growth_data'[Humidity] < 50, "Dry",    'plant_growth_data'[Humidity] <= 70, "Moderate",    "Humid")Temperature_Range_Description = SWITCH(    TRUE(),    'plant_growth_data'[Temperature] < 15, "Very Cold",    'plant_growth_data'[Temperature] < 20, "Cold",    'plant_growth_data'[Temperature] <= 25, "Cool",    'plant_growth_data'[Temperature] <= 30, "Moderate",    'plant_growth_data'[Temperature] <= 35, "Warm",    "Hot")Growth_Milestone_Description = SWITCH(    'plant_growth_data'[Growth_Milestone],    "germination", "Stage 1: Germination",    "vegetative", "Stage 2: Vegetative Growth",    "flowering", "Stage 3: Flowering",    "fruiting", "Stage 4: Fruiting",    "mature", "Stage 5: Mature",    "Unknown Stage")Plant_Growth_Category = IF(    AND(        'plant_growth_data'[Soil_Type] = "loam",        'plant_growth_data'[Water_Frequency] = "daily",        OR(            'plant_growth_data'[Fertilizer_Type] = "chemical",            'plant_growth_data'[Fertilizer_Type] = "organic"        ),        'plant_growth_data'[Temperature] >= 20,        'plant_growth_data'[Temperature] <= 30,        'plant_growth_data'[Humidity] >= 50,        'plant_growth_data'[Humidity] <= 70    ),    "Optimal Growth",    "Standard Growth")
